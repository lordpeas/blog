### 快速排序
``` js
var arr = [2, 8, 6, 4, 7, 3, 5];
function quickSort(arr){
  if(arr.length<=1){return arr};
  var middleIndex = Math.floor(arr.length/2);
  var middleItem = arr.splice(middleIndex, 1);
  var left = [], right = [];
  for(var i = 0; i < arr.length; i++){
    if(arr[i]<middleItem){
      left.push(arr[i]);     
    }else{
      right.push(arr[i]);
    }
  }
  return quickSort(left).concat(middleItem, quickSort(right));
}
var resArr = quickSort(arr);
console.log(resArr);   // [2, 3, 4, 5, 6, 7, 8]
```

### 多叉树 - 深度遍历（DFS）
``` js
var tree = {
  value: 2,
  children: [{
      value: 4,
      children: [{
          value: 5,
        }, {
          value: 7,
        }, {
          value: 8,
        },
      ]},{
        value: 4,
      },{
        value: 8,
        children: [{
          value: 10
        }]
      }
   ]
}

var resArr = [];
function deepTree(tree){
  if(tree.value){
    resArr.push(tree.value);
    if(tree.children){
      tree.children.forEach(function(item, index){
        deepTree(item);
      })
    }
  }
}

deepTree(tree);
console.log(resArr);
```

### 树 - 广度遍历
``` js
var tree = {
  value: 2,
  children: [{
      value: 4,
      children: [{
          value: 5,
        }, {
          value: 7,
        }, {
          value: 8,
        },
      ]},{
        value: 4,
      },{
        value: 8,
        children: [{
          value: 10
        }]
      }
   ]
}

var resArr = [];
function bfsTree(tree){
  if(tree.value){
    var _queue = [];
    _queue.unshift(tree);
    while(_queue.length > 0){
      var curItem = _queue.shift();
      resArr.push(curItem.value);
      if(curItem.children){
          _queue = _queue.concat(curItem.children)
      }
    }
  }
}
bfsTree(tree);
console.log(resArr);
```
